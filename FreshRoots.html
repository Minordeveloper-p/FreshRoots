import React, { useState, useEffect } from "react";

import { useRouter } from "next/router";

import { Button } from "@/components/ui/button";

import { Card, CardContent } from "@/components/ui/card";

import { Input } from "@/components/ui/input";


const initialProducts = [

  { name: "Organic Tomatoes", desc: "Fresh, seasonal, and local." },

  { name: "Farm Fresh Bananas", desc: "Sweet, ripe, and pesticide-free." },

  { name: "Green Spinach", desc: "Leafy, nutritious, and local." },

];


export default function FreshRootsHome() {

  const [products, setProducts] = useState(initialProducts);

  const [email, setEmail] = useState("");

  const [formStatus, setFormStatus] = useState<"idle" | "success" | "error">("idle");

  const router = useRouter();


  // Simulate fetching products from an API

  useEffect(() => {

    setTimeout(() => {

      setProducts([

        ...initialProducts,

        { name: "Fresh Carrots", desc: "Crunchy and sweet, direct from farms." },

        { name: "Juicy Mangoes", desc: "Seasonal delight, organically grown." },

      ]);

    }, 1000);

  }, []);


  const handleFormSubmit = (e: React.FormEvent) => {

    e.preventDefault();

    if (email) {

      setFormStatus("success");

      setEmail("");

    } else {

      setFormStatus("error");

    }

    setTimeout(() => setFormStatus("idle"), 3000);

  };


  return (

    <div className="min-h-screen bg-[#F7F5F2] text-[#2F2F2F]">

      <section className="bg-green-100 p-8 text-center">

        <h1 className="text-4xl font-bold text-green-800">FreshRoots</h1>

        <p className="mt-2 text-xl text-green-700">Rooted in Nature, Delivered Fresh</p>

        <div className="mt-6 space-x-4">

          <Button

            className="bg-green-600 hover:bg-green-700 text-white"

            onClick={() => router.push("/order")}

          >

            Order Now

          </Button>

          <Button

            variant="outline"

            onClick={() => router.push("/vendor")}

          >

            Join as Vendor

          </Button>

        </div>

      </section>


      <section className="p-8 max-w-4xl mx-auto">

        <h2 className="text-2xl font-semibold mb-4">About Us</h2>

        <p>

          FreshRoots is your trusted partner for 100% organic and pesticide-free fruits and

          vegetables, connecting Indian farmers with local vendors and households. We believe

          in clean eating, sustainability, and fair trade.

        </p>

      </section>


      <section className="bg-white p-8">

        <h2 className="text-2xl font-semibold mb-6 text-center">How It Works</h2>

        <div className="grid grid-cols-1 md:grid-cols-3 gap-6 text-center">

          <Card>

            <CardContent className="p-4">

              <h3 className="font-bold text-lg mb-2">1. Sourced from Organic Farms</h3>

              <p>We partner with certified organic farmers for pesticide-free produce.</p>

            </CardContent>

          </Card>

          <Card>

            <CardContent className="p-4">

              <h3 className="font-bold text-lg mb-2">2. Sorted & Packed Fresh</h3>

              <p>All items are quality checked, cleaned, and packed with care.</p>

            </CardContent>

          </Card>

          <Card>

            <CardContent className="p-4">

              <h3 className="font-bold text-lg mb-2">3. Delivered to You</h3>

              <p>We deliver to vendors and homes across the city every day.</p>

            </CardContent>

          </Card>

        </div>

      </section>


      <section className="p-8 max-w-4xl mx-auto">

        <h2 className="text-2xl font-semibold mb-4">Popular Products</h2>

        <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 gap-4">

          {products.map((item, index) => (

            <Card key={index}>

              <CardContent className="p-4">

                <h3 className="font-semibold">{item.name}</h3>

                <p className="text-sm text-gray-500">{item.desc}</p>

              </CardContent>

            </Card>

          ))}

        </div>

      </section>


      <section className="bg-green-50 p-8 text-center">

        <h2 className="text-2xl font-semibold mb-4">Get in Touch</h2>

        <p className="mb-4">Interested in partnering with us or ordering? Drop us a message!</p>

        <form className="flex justify-center gap-4 max-w-md mx-auto" onSubmit={handleFormSubmit}>

          <Input

            type="email"

            placeholder="Your Email"

            className="flex-1"

            required

            value={email}

            onChange={e => setEmail(e.target.value)}

          />

          <Button type="submit" className="bg-green-600 text-white">Send</Button>

        </form>

        {formStatus === "success" && (

          <p className="mt-2 text-green-700">Thank you! We'll be in touch soon.</p>

        )}

        {formStatus === "error" && (

          <p className="mt-2 text-red-600">Please enter a valid email.</p>

        )}

      </section>


      <footer className="bg-green-800 text-white p-4 text-center">

        <p>© 2025 FreshRoots. All rights reserved.</p>

      </footer>

    </div>

  );

}
